#include 
dht DHT;
#define DHT11_PIN 7
int motorPin = 3; // set pin 3 as motorPin

void setup(){
  pinMode(3, OUTPUT); // pin 3 as output
  analogWrite(3, 0); // pin 3 at LOW/OFF state initially.
}

void loop(){
  int chk = DHT.read11(DHT11_PIN); // read the temperature
  // open the fan automatically based on the detected temperature
  if (DHT.temperature >= 27 ){ // if temperature is greater than 27 celsius
    analogWrite(motorPin, 400); // turn on the motor at medium level.
  } else if (DHT.temperature >= 30){ // if temperature is greater than 30 celsius
    analogWrite(motorPin, 800); // turn on the motor at high level.
  } else { // if temperature is lower than 27 celsius
    analogWrite(motorPin, 0); // turn off the motor.
  }
}
